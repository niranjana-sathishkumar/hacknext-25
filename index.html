--- REQUIRED MODULES --- const express = require('express'); const mongoose = require('mongoose'); const bcrypt = require('bcryptjs'); const jwt = require('jsonwebtoken'); const { GoogleGenAI } = require('@google/genai'); const cors = require('cors'); require('dotenv').config(); // Load environment variables from .env file // --- CONFIGURATION --- const app = express(); const PORT = process.env.PORT || 3000; const JWT_SECRET = process.env.JWT_SECRET; const MONGO_URI = process.env.MONGO_URI; // Initialize Google GenAI const ai = new GoogleGenAI({ apiKey: process.env.GEMINI_API_KEY }); const AI_MODEL = "gemini-2.5-flash-preview-09-2025"; // --- MIDDLEWARE --- app.use(cors()); // Enable CORS for client-server communication app.use(express.json()); // Body parser for JSON requests // --- DATABASE CONNECTION (MongoDB) --- mongoose.connect(MONGO_URI) .then(() => console.log('MongoDB Connected Successfully')) .catch(err => { console.error('MongoDB Connection Error:', err); process.exit(1); // Exit process if connection fails }); // --- MONGOOSE USER MODEL --- const UserSchema = new mongoose.Schema({ email: { type: String, required: true, unique: true }, password: { type: String, required: true }, createdAt: { type: Date, default: Date.now } }); const User = mongoose.model('User', UserSchema); // --- AUTHENTICATION MIDDLEWARE --- const verifyToken = (req, res, next) => { // Get token from header const token = req.header('Authorization')?.replace('Bearer ', ''); if (!token) { return res.status(401).json({ msg: 'Access denied. No token provided.' }); } try { // Verify token const decoded = jwt.verify(token, JWT_SECRET); req.user = decoded.user; next(); } catch (e) { res.status(401).json({ msg: 'Token is not valid or expired.' }); } }; // --- AUTH ROUTES --- // @route POST /signup // @desc Register new user app.post('/signup', async (req, res) => { const { email, password } = req.body; if (!email || !password || password.length < 6) { return res.status(400).json({ error: 'Please enter a valid email and a password of at least 6 characters.' }); } try { // Check if user exists let user = await User.findOne({ email }); if (user) { return res.status(400).json({ error: 'User already exists.' }); } // Create new user instance user = new User({ email, password }); // Hash password const salt = await bcrypt.genSalt(10); user.password = await bcrypt.hash(password, salt); await user.save(); // Create JWT payload const payload = { user: { id: user.id } }; // Sign token and send back jwt.sign(payload, JWT_SECRET, { expiresIn: '1h' }, (err, token) => { if (err) throw err; res.status(201).json({ success: true, token, user: { email: user.email, id: user.id } }); }); } catch (err) { console.error('Signup Error:', err.message); res.status(500).json({ error: 'Server error during registration.' }); } }); // @route POST /login // @desc Authenticate user & get token app.post('/login', async (req, res) => { const { email, password } = req.body; if (!email || !password) { return res.status(400).json({ error: 'Please enter both email and password.' }); } try { let user = await User.findOne({ email }); if (!user) { return res.status(400).json({ error: 'Invalid Credentials.' }); } const isMatch = await bcrypt.compare(password, user.password); if (!isMatch) { return res.status(400).json({ error: 'Invalid Credentials.' }); } // Create JWT payload const payload = { user: { id: user.id } }; // Sign token and send back jwt.sign(payload, JWT_SECRET, { expiresIn: '1h' }, (err, token) => { if (err) throw err; res.json({ success: true, token, user: { email: user.email, id: user.id } }); }); } catch (err) { console.error('Login Error:', err.message); res.status(500).json({ error: 'Server error during login.' }); } }); // --- GEMINI API ROUTES (Protected) --- // Helper function for API calls with backoff async function callGeminiApiWithBackoff(func) { let delay = 1000; for (let i = 0; i < 3; i++) { try { return await func(); } catch (error) { if (i === 2) throw error; // Re-throw on the last attempt console.warn(API call failed, retrying in ${delay}ms..., error.message); await new Promise(resolve => setTimeout(resolve, delay)); delay *= 2; } } } // @route POST /generate-curriculum // @desc Generates a 30-day curriculum based on user input app.post('/generate-curriculum', verifyToken, async (req, res) => { const { subject, style, quizResults } = req.body; // Detailed prompt for the Gemini Model // IMPORTANT: The placeholder values [day_number] and [topic_slug] must be literal text for the AI to parse, // NOT JavaScript template literals (like ${day_number}), which caused the previous error. const systemPrompt = `You are an expert curriculum designer. Generate a structured, 30-day learning curriculum tailored to a user's subject, learning style, and prior knowledge. The output MUST be a strict JSON array of 30 objects, one for each day. Each object MUST have the following keys: 1. "day": (Number) The day number (1 to 30). 2. "topic": (String) The specific concept for the day. 3. "explanation_summary": (String) A concise, personalized explanation of the concept, adapting the language and examples to the requested learning style ("${style}"). 4. "practice_problem": (String) A challenging problem or task related to the day's topic. 5. "content_link": (String) A placeholder for a suggested external resource (e.g., "https://example.com/search?q=day[day_number]+[topic_slug]"). The curriculum must be progressive, building on prior days' concepts. Prior Knowledge: The user scored ${quizResults.correct}/${quizResults.total} on the quiz, indicating a need to focus on foundational topics first if the score is low.`; const userQuery = Create the 30-day curriculum for the subject: ${subject}. Learning Style: ${style}.; try { const generationFunc = () => ai.models.generateContent({ model: AI_MODEL, contents: [{ parts: [{ text: userQuery }] }], config: { systemInstruction: { parts: [{ text: systemPrompt }] }, responseMimeType: "application/json", responseSchema: { type: "ARRAY", items: { type: "OBJECT", properties: { "day": { "type": "INTEGER" }, "topic": { "type": "STRING" }, "explanation_summary": { "type": "STRING" }, "practice_problem": { "type": "STRING" }, "content_link": { "type": "STRING" } }, required: ["day", "topic", "explanation_summary", "practice_problem", "content_link"] } } } }); const response = await callGeminiApiWithBackoff(generationFunc); const jsonText = response.candidates[0].content.parts[0].text; const curriculum = JSON.parse(jsonText); res.json({ success: true, curriculum }); } catch (error) { console.error('Curriculum Generation Error:', error); res.status(500).json({ error: 'Failed to generate curriculum due to an AI service error.' }); } }); // @route POST /chat // @desc Handles the AI chatbot conversation app.post('/chat', verifyToken, async (req, res) => { const { subject, history } = req.body; const systemPrompt = You are an AI Study Partner specializing in ${subject}. Your goal is to provide concise, helpful, and grounded answers. Respond as a supportive tutor, explaining concepts clearly and referring to the user's curriculum if contextually relevant. Use Google Search grounding for up-to-date and factual information.; const userQuery = history[history.length - 1].parts[0].text; // Get the latest message try { const generationFunc = () => ai.models.generateContent({ model: AI_MODEL, contents: [{ parts: [{ text: userQuery }] }], config: { systemInstruction: { parts: [{ text: systemPrompt }] }, tools: [{ google_search: {} }] // Enable Google Search grounding } }); const response = await callGeminiApiWithBackoff(generationFunc); const candidate = response.candidates?.[0]; const text = candidate?.content?.parts?.[0]?.text || "I'm sorry, I couldn't process that request."; // Extract sources let sources = []; const groundingMetadata = candidate?.groundingMetadata; if (groundingMetadata && groundingMetadata.groundingAttributions) { sources = groundingMetadata.groundingAttributions .map(attribution => ({ uri: attribution.web?.uri, title: attribution.web?.title, })) .filter(source => source.uri && source.title); } res.json({ success: true, response: text, sources }); } catch (error) { console.error('Chat API Error:', error); res.status(500).json({ error: 'Failed to get a response from the AI model.' }); } }); // --- SERVER START --- app.listen(PORT, () => { console.log(Server running on http://localhost:${PORT}); console.log('Ensure you run "npm install express mongoose bcryptjs jsonwebtoken dotenv @google/genai cors" first.'); });
